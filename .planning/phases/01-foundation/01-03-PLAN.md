  CMDB_IMPORT: 'cmdb.import',
} as const;

export function hasPermission(userPermissions: string[], required: string): boolean {
  if (userPermissions.includes('*')) return true;
  if (userPermissions.includes(required)) return true;
  // Check wildcard: 'tickets.*' matches 'tickets.read'
  const parts = required.split('.');
  for (let i = parts.length - 1; i > 0; i--) {
    const wildcard = parts.slice(0, i).join('.') + '.*';
    if (userPermissions.includes(wildcard)) return true;
  }
  return false;
}
```

**4. Auth service (`apps/api/src/services/auth.service.ts`):**
- `validateCredentials(email, password, tenantId)`: Look up user by email + tenantId, verify password with `@node-rs/bcrypt` `verifySync`, return user or null.
- `getUserRoles(userId, tenantId)`: Query UserRole + Role join to get all role slugs and permissions for a user.
- `generateTokens(payload, fastify)`: Sign access token (15min expiry) and refresh token (7 day expiry) using `fastify.jwt.sign()`. Access token payload: `{ userId, tenantId, email, roles, type: 'access' }`. Refresh token payload: `{ userId, tenantId, email, roles, type: 'refresh' }`.
- `createPasswordResetToken(userId, tenantId)`: Generate random token, hash with SHA-256, store in PasswordResetToken table with 1-hour expiry. Return the unhashed token (sent via email).
- `validatePasswordResetToken(token)`: Hash incoming token, find in DB, check not expired and not used.
- `resetPassword(token, newPassword)`: Validate token, hash new password, update user, mark token as used.

**5. User service (`apps/api/src/services/user.service.ts`):**
- `findById(userId, tenantId)`: Return user with roles.
- `findByEmail(email, tenantId)`: Return user with roles.

`plugins/swagger.ts`: Register `@fastify/swagger` with OpenAPI 3.1 info (title: "MeridianITSM API", version: "1.0.0").

`plugins/auth.ts`: `authPreHandler` — calls `request.jwtVerify()`. On failure, replies 401 `{ error: 'Unauthorized' }`.

`plugins/tenant.ts`: `tenantPreHandler` — extracts `tenantId` from `request.user.tenantId` (set by JWT verify). Looks up tenant in DB using raw prisma (not scoped — Tenant is a global model). If tenant not found or status is not ACTIVE, replies 403 `{ error: 'Tenant not found or suspended' }`. Sets `request.tenant` and `request.tenantId`. Also loads the full user record with roles and sets `request.currentUser`.

`plugins/rbac.ts`: `requirePermission(permission)` — returns a preHandler function that checks `hasPermission(request.currentUser.roles, permission)`. If not permitted, replies 403 `{ error: 'Forbidden', message: 'Missing permission: {permission}' }`. Also export `rbacPreHandler` as a no-op passthrough (permissions checked per-route, not globally).

`plugins/api-key.ts`: `apiKeyPreHandler` — checks `Authorization: ApiKey <key>` header. Hashes key with SHA-256, looks up in ApiKey table by keyHash where isActive=true and (expiresAt is null OR expiresAt > now). If valid, sets `request.tenantId`, `request.apiKey` with id, scopes, tenantId. Looks up tenant to verify it's ACTIVE. If invalid, replies 401.

**7. Routes:**

`routes/health/index.ts`: Fastify plugin that registers `GET /api/health` returning `{ status: 'ok', timestamp: new Date().toISOString(), version: '1.0.0' }`.

`routes/auth/index.ts`: Fastify plugin that registers all auth sub-routes under `/api/auth`. Apply `AUTH_RATE_LIMIT` to login and password-reset routes.

`routes/auth/login.ts`: `POST /api/auth/login` — accepts `{ email, password, tenantSlug }` (tenantSlug used to look up the tenant first since we don't have JWT yet). Validates with Zod loginSchema. Calls `authService.validateCredentials`. On success, generates tokens and returns `{ accessToken, refreshToken, user: { id, email, firstName, lastName, roles } }`. On failure, returns 401 `{ error: 'Invalid credentials' }`.

`routes/auth/refresh.ts`: `POST /api/auth/refresh` — accepts `{ refreshToken }`. Verifies the refresh JWT, checks type is 'refresh'. Generates new access + refresh tokens. Returns same shape as login.

`routes/auth/password-reset.ts`: Two endpoints:
- `POST /api/auth/password-reset/request` — accepts `{ email }`. Always returns 200 (don't leak whether email exists). If user exists, creates reset token and logs it (email sending wired in Phase 3).
- `POST /api/auth/password-reset/reset` — accepts `{ token, password }`. Validates token, resets password. Returns 200 `{ message: 'Password reset successful' }` or 400 on invalid/expired token.

import { prisma } from '@meridian/db';
import { NextResponse } from 'next/server';

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const subdomain = searchParams.get('subdomain');

  if (!subdomain) {
    return NextResponse.json({ error: 'subdomain parameter required' }, { status: 400 });
  }

  const tenant = await prisma.tenant.findFirst({
    where: { subdomain, status: 'ACTIVE' },
    select: { id: true, name: true, subdomain: true, backendUrl: true },
  });

  if (!tenant) {
    return NextResponse.json({ error: 'Tenant not found' }, { status: 404 });
  }

  return NextResponse.json({
    tenantId: tenant.id,
    name: tenant.name,
    backendUrl: tenant.backendUrl ?? `http://localhost:4000`,
  });
}
```
