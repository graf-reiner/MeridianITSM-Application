    **Per-tenant loop** body — full implementation per RESEARCH lines 261-326:
    ```typescript
    interface TenantResult {
      tenantId: string; tenantName: string;
      assetsProcessed: number;
      ciExtUpserted: number; ciAutoCreated: number;
      softwareUpserted: number;
      conflictsLogged: number;
      unparseableSoftware: number;
    }

    async function migrateTenant(tenantId: string, tenantName: string): Promise<TenantResult> {
      console.log(`\n=== Tenant: ${tenantName} (${tenantId}) ===`);
      let assetsProcessed = 0, ciExtUpserted = 0, ciAutoCreated = 0;
      let softwareUpserted = 0, conflictsLogged = 0, unparseableSoftware = 0;

      // Step 1: read all Assets with any hardware data via raw SQL (Pitfall 1)
      const candidates = await prisma.$queryRaw<Array<{
        id: string; hostname: string | null; operatingSystem: string | null;
        osVersion: string | null; cpuModel: string | null; cpuCores: number | null;
        ramGb: number | null; disks: unknown; networkInterfaces: unknown;
        softwareInventory: unknown; lastInventoryAt: Date | null;
      }>>`
        SELECT id, hostname, "operatingSystem", "osVersion", "cpuModel",
               "cpuCores", "ramGb", disks, "networkInterfaces",
               "softwareInventory", "lastInventoryAt"
          FROM "assets"
         WHERE "tenantId" = ${tenantId}::uuid
           AND (hostname IS NOT NULL
                OR "operatingSystem" IS NOT NULL
                OR "osVersion" IS NOT NULL
                OR "cpuModel" IS NOT NULL
                OR "cpuCores" IS NOT NULL
                OR "ramGb" IS NOT NULL
                OR disks IS NOT NULL
                OR "networkInterfaces" IS NOT NULL
                OR "softwareInventory" IS NOT NULL
                OR "lastInventoryAt" IS NOT NULL)`;

      console.log(`  Found ${candidates.length} Assets with hardware data to migrate`);

      // Step 2: per-Asset migration in own transaction (advisory-lock per Pitfall 2)
      for (const asset of candidates) {
        await prisma.$transaction(async (tx) => {
          assetsProcessed++;

          // Acquire same advisory lock the live createCI uses
          await tx.$executeRaw`SELECT pg_advisory_xact_lock(hashtext(${tenantId} || '_ci_seq'))`;

          // Find or create linked CI
          let ciId: string;
          const linked = await tx.$queryRaw<Array<{ id: string }>>`
            SELECT id FROM "cmdb_configuration_items"
             WHERE "tenantId" = ${tenantId}::uuid AND "assetId" = ${asset.id}::uuid
             ORDER BY "createdAt" ASC LIMIT 1`;

          if (linked.length === 0) {
            // Orphan path — auto-create CI
            const { classKey } = inferClassKeyFromSnapshot(null, asset.hostname, asset.operatingSystem);
            // Resolve refs via raw SQL (avoid importing apps/api resolvers)
            const classRow = await tx.$queryRaw<Array<{ id: string }>>`
              SELECT id FROM "cmdb_ci_classes" WHERE "tenantId" = ${tenantId}::uuid AND "classKey" = ${classKey} LIMIT 1`;
            const lifecycleRow = await tx.$queryRaw<Array<{ id: string }>>`
              SELECT id FROM "cmdb_lifecycle_statuses" WHERE "tenantId" = ${tenantId}::uuid AND "statusKey" = 'in_service' LIMIT 1`;
            const operationalRow = await tx.$queryRaw<Array<{ id: string }>>`
              SELECT id FROM "cmdb_operational_statuses" WHERE "tenantId" = ${tenantId}::uuid AND "statusKey" = 'online' LIMIT 1`;
            const envRow = await tx.$queryRaw<Array<{ id: string }>>`
              SELECT id FROM "cmdb_environments" WHERE "tenantId" = ${tenantId}::uuid AND "envKey" = 'prod' LIMIT 1`;

            if (!classRow[0] || !lifecycleRow[0] || !operationalRow[0] || !envRow[0]) {
              throw new Error(`Phase 8 backfill: tenant ${tenantId} missing reference data — run seed-existing-tenants-cmdb-ref.ts`);
            }

            // Allocate ciNumber under same lock
            const next = await tx.$queryRaw<Array<{ next: number }>>`
              SELECT COALESCE(MAX("ciNumber"), 0) + 1 AS next FROM "cmdb_configuration_items" WHERE "tenantId" = ${tenantId}::uuid`;
            const ciNumber = Number(next[0]?.next ?? 1);

            const created = await tx.cmdbConfigurationItem.create({
              data: {
                tenantId, classId: classRow[0].id,
                lifecycleStatusId: lifecycleRow[0].id,
                operationalStatusId: operationalRow[0].id,
                environmentId: envRow[0].id,
                ciNumber, name: asset.hostname || `unnamed-asset-${asset.id.slice(0, 8)}`,
                assetId: asset.id,
              },
              select: { id: true },
            });
            ciId = created.id;
            ciAutoCreated++;
          } else {
            ciId = linked[0].id;
          }

          // Read existing CmdbCiServer extension (if any) for conflict detection
          const existingExt = await tx.cmdbCiServer.findUnique({ where: { ciId } });

          // Build conflict audit rows + extension write data (CI wins per D-01)
          const auditRows: Prisma.CmdbMigrationAuditCreateManyInput[] = [];
          const extWriteData: Record<string, unknown> = {};

          // map asset fields → ext fields with conflict logging
          const fieldPairs: Array<[string, unknown, unknown]> = [
            ['operatingSystem', existingExt?.operatingSystem, asset.operatingSystem],
            ['osVersion', existingExt?.osVersion, asset.osVersion],
            ['cpuCount', existingExt?.cpuCount, asset.cpuCores],
            ['cpuModel', existingExt?.cpuModel, asset.cpuModel],
            ['memoryGb', existingExt?.memoryGb, asset.ramGb],
            ['disksJson', existingExt?.disksJson, asset.disks],
            ['networkInterfacesJson', existingExt?.networkInterfacesJson, asset.networkInterfaces],
          ];
          for (const [field, ciValue, assetValue] of fieldPairs) {
            if (assetValue == null) continue;
            if (ciValue == null) {
              // CI lacks the value, Asset has it → write to CI
              extWriteData[field] = assetValue;
            } else if (String(JSON.stringify(ciValue)) !== String(JSON.stringify(assetValue))) {
              // Both have values that differ → CI wins, log conflict
              auditRows.push({
                tenantId, tableName: 'assets', recordId: asset.id,
                fieldName: field,
                oldValue: String(JSON.stringify(assetValue)).slice(0, 1000),
                newValue: String(JSON.stringify(ciValue)).slice(0, 1000),
                status: 'overwritten_by_ci', phase: 'phase8',
              });
            }
            // else: equal — noop
          }

          // Upsert CmdbCiServer extension
          if (!DRY_RUN) {
            await tx.cmdbCiServer.upsert({
              where: { ciId },
              create: {
                ciId, tenantId,
                serverType: 'physical',  // unknown at backfill time; reconciliation worker can refine later
                operatingSystem: extWriteData['operatingSystem'] as string | undefined ?? null,
                osVersion: extWriteData['osVersion'] as string | undefined ?? null,
                cpuCount: extWriteData['cpuCount'] as number | undefined ?? null,
                cpuModel: extWriteData['cpuModel'] as string | undefined ?? null,
                memoryGb: extWriteData['memoryGb'] as number | undefined ?? null,
                disksJson: extWriteData['disksJson'] as never,
                networkInterfacesJson: extWriteData['networkInterfacesJson'] as never,
              },
              update: extWriteData,  // CI wins: only write fields where CI was null
            });
            ciExtUpserted++;
          }

          // Conflict audit (batched per Asset, skipDuplicates per Pitfall 4)
          if (auditRows.length > 0 && !DRY_RUN) {
            await tx.cmdbMigrationAudit.createMany({ data: auditRows, skipDuplicates: true });
            conflictsLogged += auditRows.length;
          }

          // Software: explode JSON blob (Pitfall 8 + 10)
          const softwareList = parseSoftwareList(asset.softwareInventory);
          if (softwareList.length === 0 && asset.softwareInventory != null) {
            // Unparseable
            unparseableSoftware++;
            if (!DRY_RUN) {
              await tx.cmdbMigrationAudit.create({
                data: {
                  tenantId, tableName: 'assets', recordId: asset.id,
                  fieldName: 'softwareInventory',
                  oldValue: JSON.stringify(asset.softwareInventory).slice(0, 500),
                  newValue: null, status: 'unparseable_software_blob', phase: 'phase8',
                },
              });
            }
          }
          for (const item of softwareList) {
            const normalizedVersion = (item.version ?? '').trim() || 'unknown';   // Pitfall 3
            if (!DRY_RUN) {
              await tx.cmdbSoftwareInstalled.upsert({
                where: { ciId_name_version: { ciId, name: item.name, version: normalizedVersion } },
                create: {
                  tenantId, ciId, name: item.name, version: normalizedVersion,
                  vendor: item.vendor ?? null, publisher: item.publisher ?? null,
                  installDate: item.installDate ? new Date(item.installDate) : null,
                  source: 'import', lastSeenAt: new Date(),
                },
                update: {},  // CI wins per D-01 — keep existing data; only refresh lastSeenAt
              });
              softwareUpserted++;
            }
          }
        }, { timeout: 30_000 });
      }

      console.log(`  Tenant ${tenantName}: ${assetsProcessed} assets processed, ${ciExtUpserted} ext upserts, ${ciAutoCreated} CIs auto-created, ${softwareUpserted} software rows, ${conflictsLogged} conflicts logged${unparseableSoftware ? `, ${unparseableSoftware} unparseable software blobs` : ''}`);
      return { tenantId, tenantName, assetsProcessed, ciExtUpserted, ciAutoCreated, softwareUpserted, conflictsLogged, unparseableSoftware };
    }
    ```
