    ```typescript
    export interface SoftwareInventoryReportFilters {
      softwareName?: string;
      vendor?: string;
      publisher?: string;
      ciClassKey?: string;
      page?: number;
      pageSize?: number;
    }

    /**
     * Phase 8 (CASR-03 / CRIT-5): software-by-CI listing for license reporting.
     * Tenant-scoped via tenantId on cmdb_software_installed.
     * NB: licenseKey is intentionally OMITTED from this list response —
     *     the CI-scoped /api/v1/cmdb/cis/:id/software endpoint surfaces it
     *     with cmdb.view permission.
     */
    export async function getSoftwareInventoryReport(
      tenantId: string,
      filters: SoftwareInventoryReportFilters = {},
    ): Promise<{
      data: Array<{
        ciId: string; ciName: string; ciNumber: number; classKey: string;
        name: string; version: string;
        vendor: string | null; publisher: string | null;
        lastSeenAt: Date;
      }>;
      count: number;
    }> {
      const page = filters.page ?? 1;
      const pageSize = Math.min(filters.pageSize ?? 50, 200);
      const skip = (page - 1) * pageSize;

      const where: Prisma.CmdbSoftwareInstalledWhereInput = {
        tenantId,
        ...(filters.softwareName && { name: { contains: filters.softwareName, mode: 'insensitive' as const } }),
        ...(filters.vendor && { vendor: filters.vendor }),
        ...(filters.publisher && { publisher: filters.publisher }),
        ...(filters.ciClassKey && { ci: { ciClass: { classKey: filters.ciClassKey } } }),
      };

      const [rows, count] = await Promise.all([
        prisma.cmdbSoftwareInstalled.findMany({
          where,
          select: {
            // Explicit select OMITTING licenseKey
            ciId: true, name: true, version: true,
            vendor: true, publisher: true, lastSeenAt: true,
            ci: { select: { id: true, name: true, ciNumber: true, ciClass: { select: { classKey: true } } } },
          },
          orderBy: [{ ci: { name: 'asc' } }, { name: 'asc' }],
          skip, take: pageSize,
        }),
        prisma.cmdbSoftwareInstalled.count({ where }),

    import { requirePermission } from '../../../plugins/rbac.js';
    import { prisma } from '@meridian/db';

    const patchCiSchema = z.object({
      assetId: z.string().uuid().nullable().optional(),
      // (any other CI fields the existing handler already supports — preserve them)
    }).strict();

    fastify.patch(
      '/api/v1/cmdb/cis/:id',
      { preHandler: requirePermission('cmdb.edit') },
      async (request, reply) => {
        const user = request.user as { tenantId: string; userId: string };
        const { id: ciId } = request.params as { id: string };
        const parsed = patchCiSchema.safeParse(request.body);
        if (!parsed.success) return reply.code(400).send({ error: 'Invalid body', issues: parsed.error.issues });
        const body = parsed.data;

        // Defense-in-depth: verify CI belongs to tenant
        const ci = await prisma.cmdbConfigurationItem.findFirst({
          where: { id: ciId, tenantId: user.tenantId },
          select: { id: true },
        });
        if (!ci) return reply.code(404).send({ error: 'CI not found' });

        // If linking an Asset (assetId is a non-null string), verify Asset belongs to same tenant
        if (typeof body.assetId === 'string') {
          const asset = await prisma.asset.findFirst({
            where: { id: body.assetId, tenantId: user.tenantId },
            select: { id: true },
          });
          if (!asset) return reply.code(404).send({ error: 'Asset not found in this tenant' });
        }

        // Build the update payload — only include fields actually supplied
        const updateData: Record<string, unknown> = {};
        if ('assetId' in body) updateData.assetId = body.assetId;   // null is a valid unlink

        const updated = await prisma.cmdbConfigurationItem.update({
          where: { id: ciId },
          data: updateData,
          select: { id: true, assetId: true },
        });
        return reply.send({ data: updated });
      },
    );
    ```
