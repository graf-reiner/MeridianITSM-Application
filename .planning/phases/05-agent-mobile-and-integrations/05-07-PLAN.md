1. **API hooks** — Create TanStack Query hooks using the Axios client:

   `apps/mobile/src/api/tickets.ts`:
   ```typescript
   import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
   import { apiClient } from './client';

   export function useTickets(filters?: { status?: string; page?: number }) {
     return useQuery({
       queryKey: ['tickets', filters],
       queryFn: () => apiClient.get('/api/v1/tickets', { params: filters }).then(r => r.data),
     });
   }

   export function useTicket(id: string) {
     return useQuery({
       queryKey: ['ticket', id],
       queryFn: () => apiClient.get(`/api/v1/tickets/${id}`).then(r => r.data),
       enabled: !!id,
     });
   }

   export function useCreateTicket() {
     const qc = useQueryClient();
     return useMutation({
       mutationFn: (data: { title: string; description: string; type: string; priority: string; categoryId?: string }) =>
         apiClient.post('/api/v1/tickets', data).then(r => r.data),
       onSuccess: () => qc.invalidateQueries({ queryKey: ['tickets'] }),
     });
   }

   export function useUpdateTicket() { /* PATCH /api/v1/tickets/:id */ }
   export function useAddComment() { /* POST /api/v1/tickets/:id/comments with FormData for attachments */ }
   export function useMyTickets() { /* GET /api/v1/tickets?assignedToId=me */ }
   ```
