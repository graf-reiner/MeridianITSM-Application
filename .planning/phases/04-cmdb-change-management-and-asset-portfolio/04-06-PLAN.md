Create apps/api/src/services/application.service.ts:

1. createApp(prisma, tenantId, data, userId):
   - data: { name, type?, status?, criticality?, hostingModel?, techStack?, authMethod?, dataClassification?, annualCost?, rpo?, rto?, lifecycleStage?, strategicRating?, description?, customFields? }
   - Create Application record
   - Log ApplicationActivity: activityType='CREATED', actorId=userId

2. getApp(prisma, tenantId, appId):
   - Include: dependencies (sourceApp with target app name/type/status), dependents (targetApp with source app name/type/status), documents, applicationAssets (with asset details), activities (limit 50, ordered by createdAt DESC)

3. listApps(prisma, tenantId, filters):
   - Filters: type?, status?, criticality?, hostingModel?, lifecycleStage?, search? (name)
   - Pagination: page, pageSize
   - Return { data: Application[], total, page, pageSize }

4. updateApp(prisma, tenantId, appId, data, userId):
   - Compare changed fields, log ApplicationActivity for each changed field with oldValue/newValue
   - Update application

5. deleteApp(prisma, tenantId, appId):
   - Delete related dependencies (both source and target), documents, applicationAssets, activities
   - Delete application

6. addDependency(prisma, tenantId, sourceAppId, targetAppId, dependencyType, description?):
   - Prevent self-dependency (sourceAppId !== targetAppId)
   - Create ApplicationDependency. Unique constraint prevents duplicates.

7. removeDependency(prisma, tenantId, dependencyId):
   - Delete by id + tenantId

8. addDocument(prisma, tenantId, appId, data):
   - data: { title, documentType, url, description? }
   - Create ApplicationDocument

9. removeDocument(prisma, tenantId, documentId):
   - Delete by id + tenantId

10. linkAsset(prisma, tenantId, appId, assetId, relationshipType, isPrimary?):
    - Create ApplicationAsset. Unique constraint [applicationId, assetId] prevents duplicates.

11. unlinkAsset(prisma, tenantId, appAssetId):
    - Delete by id + tenantId

12. getPortfolioStats(prisma, tenantId):
    - Total applications count
    - Count by status (ACTIVE, IN_DEVELOPMENT, PLANNED, etc.)
    - Count by criticality (LOW, MEDIUM, HIGH, CRITICAL)
    - Count deprecated/decommissioned
    - Total annual cost (SUM annualCost where status=ACTIVE)
    - Return { total, byStatus: Record, byCriticality: Record, deprecatedCount, totalAnnualCost }

13. getDependencyGraph(prisma, tenantId):
    - Fetch all applications (id, name, type, status, criticality) as nodes
    - Fetch all ApplicationDependency records as edges (sourceId, targetId, dependencyType)
    - Return { nodes: AppNode[], edges: AppEdge[] } — ready for ReactFlow rendering
